# Определение переменных
DEFINE pi 3.14159
DEFINE e 2.71828

# Пример с комментарием
# Сложение чисел
PUSH 10
PUSH 5
+
PRINT   # Должно вывести 15

# Использование переменной
PUSH pi
PUSH 2
*
PRINT   # Должно вывести примерно 6.28318

# Вычисление квадратного корня
PUSH 16
SQRT
PRINT   # Должно вывести 4

# Пример из задания
DEFINE a 4
PUSH a
SQRT
PRINT   # Должно вывести 2

# Более сложный пример с делением и умножением
PUSH 10
PUSH 2
/       # 10/2 = 5
PUSH 3
*       # 5*3 = 15
PRINT   # Должно вывести 15

# Пример с цепочкой операций
PUSH 5
PUSH 3
+       # 5+3 = 8
PUSH 2
*       # 8*2 = 16
PRINT   # Должно вывести 16

# Проверка разных арифметических операций
PUSH 20
PUSH 5
-       # 20-5 = 15
PRINT   # Должно вывести 15
